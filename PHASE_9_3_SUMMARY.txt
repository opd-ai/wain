================================================================================
PLATFORM-SPECIFIC CHALLENGES
================================================================================

Wayland:
  • Fixed-point coordinates: multiply by 1/256 to get pixels
  • Events grouped in frames: need buffering/batching
  • Separate focus enter/leave events: update focused widget separately
  • Key repeat: Need rate + delay from HandleRepeatInfo
  • Scroll via axis events: Not separate button events
  • Modifier state: Three masks (depressed, latched, locked) need combination
  • Keysym lookup: Use existing keymap.KeycodeToKeysym

X11:
  • Raw keycodes: Need XKB lookup (not currently available)
  • Scroll via buttons: Buttons 4 & 5 = scroll up/down (need mapping)
  • Modifier state: Single mask with shift/ctrl/alt/mod1-5/buttons
  • No frame grouping: Events sent independently
  • No key repeat info: Need client-side timer
  • Window manager integration: Handle WM_DELETE_WINDOW properly

PUBLIC API DESIGN (TO BE CREATED)
================================================================================

File: /event.go

type KeyEvent struct {
    Key string           // "Tab", "Escape", "a", "Enter", etc.
    Action KeyAction     // Press, Release, Repeat
    Modifiers ModifierSet
    Time uint32          // Milliseconds
    RepeatCount uint32   // For repeat action
    Consumed bool        // Set by handlers to stop propagation
}

type PointerEvent struct {
    X, Y int              // Pixel coordinates
    Button int            // 1=left, 2=middle, 3=right
    ScrollDelta float64   // For scroll action
    Action PointerAction  // Move, Press, Release, Scroll
    Modifiers ModifierSet
    Time uint32           // Milliseconds
    Consumed bool
}

type TouchEvent struct {
    ID int32              // Touch point ID
    X, Y float64          // Coordinates in fixed-point
    Action TouchAction    // Down, Up, Motion, Cancel
    Modifiers ModifierSet
    Time uint32
    Consumed bool
}

type WindowEvent struct {
    Type WindowEventType  // Resize, Close, Focus, ScaleChange
    // Data varies by type (struct could use union pattern)
}

type CustomEvent struct {
    Type string           // App-defined event type
    Data interface{}      // App-defined data
}

type KeyAction int
const (
    KeyPress   KeyAction = iota
    KeyRelease
    KeyRepeat
)

type PointerAction int
const (
    PointerMove   PointerAction = iota
    PointerPress
    PointerRelease
    PointerScroll
)

type ModifierSet struct {
    Shift, Control, Alt, Super bool
    CapsLock, NumLock bool
}

Example Usage:
  win.OnKeyPress(func(e KeyEvent) bool {
    if e.Key == "Tab" {
      focusManager.MoveFocus(true)
      return true  // consumed
    }
    if e.Key == "Escape" {
      app.Quit()
      return true
    }
    return false  // not consumed, propagate
  })

  win.OnPointerClick(func(e PointerEvent) bool {
    widget := hitTest(e.X, e.Y)
    if widget != nil {
      return widget.Handle(&e)  // delegate
    }
    return false
  })
